🔧 Recommended Fixes

1. Settings Page - Two Column Layout (HIGH PRIORITY)

/* Fix: Add breakpoint for iPhone/Samsung viewports */ /* File: settings.component.scss */ @media (max-width: 430px) { .form-row.two-columns { flex-direction: column; gap: var(--space-4); .p-field { width: 100%; flex: 1 1 100%; } } }

2. Settings Navigation Buttons (HIGH PRIORITY)

/* Fix: Stack nav items on narrow screens */ /* File: settings.component.scss */ @media (max-width: 414px) { .settings-nav { flex-direction: column; gap: var(--space-2); } .settings-nav-item { width: 100%; justify-content: flex-start; } }

3. Dialog Width Mobile Override (MEDIUM PRIORITY)

/* Fix: Full-width dialogs on narrow screens */ /* File: _mobile-responsive.scss */ @media (max-width: 430px) { .p-dialog { width: 100vw !important; max-width: 100vw !important; margin: 0; border-radius: var(--radius-lg) var(--radius-lg) 0 0; } .p-dialog-content { padding: var(--space-4); padding-bottom: max(var(--space-4), env(safe-area-inset-bottom)); } }

4. Add Device-Specific Breakpoints

/* Add to _mobile-responsive.scss */ /* iPhone 11-14 (414px viewport) */ @mixin iphone-standard { @media (max-width: 414px) { @content; } } /* iPhone 15-17 Pro Max (430px viewport) */ @mixin iphone-pro-max { @media (max-width: 430px) { @content; } } /* Samsung Galaxy S23/S24 (412px viewport) */ @mixin samsung-standard { @media (max-width: 412px) { @content; } }

CPU iPhone OS 18_0 like Mac OS X)' }, 'Samsung Galaxy S23': { viewport: { width: 412, height: 915 }, deviceScaleFactor: 2.625, userAgent: 'Mozilla/5.0 (Linux; Android 15)' }, 'Samsung Galaxy S24 Ultra': { viewport: { width: 412, height: 915 }, deviceScaleFactor: 3, userAgent: 'Mozilla/5.0 (Linux; Android 15)' }, }; // Pages to test const pagesToTest = [ { name: 'Settings', path: '/settings', requiresAuth: true }, { name: 'Login', path: '/login', requiresAuth: false }, { name: 'Dashboard', path: '/dashboard', requiresAuth: true }, { name: 'Training', path: '/training', requiresAuth: true }, ]; test.describe('Mobile Responsiveness - Visual Regression', () => { for (const [deviceName, deviceConfig] of Object.entries(targetDevices)) { test.describe(`${deviceName}`, () => { test.use({ viewport: deviceConfig.viewport, deviceScaleFactor: deviceConfig.deviceScaleFactor, userAgent: deviceConfig.userAgent, hasTouch: true, isMobile: true, }); await p.fill('input[type="password"]', process.env.TEST_PASSWORD || 'password123'); await p.click('button[type="submit"]'); await p.waitForURL('**/dashboard**', { timeout: 30000 }); } await p.goto(page.path); await p.waitForLoadState('networkidle'); // Visual snapshot await expect(p).toHaveScreenshot( `${page.name.toLowerCase()}-${deviceName.replace(/\s/g, '-').toLowerCase()}.png`, { fullPage: true, maxDiffPixelRatio: 0.05 } ); }); } }); } }); test.describe('Settings Page - Specific Mobile Tests', () => { test.use({ viewport: { width: 414, height: 896 }, hasTouch: true, isMobile: true, }); test('two-column layout stacks on mobile', async ({ page }) => { await page.goto('/settings'); // Open change password dialog await page.click('text=Change'); await page.waitForSelector('.p-dialog'); const dialog = page.locator('.p-dialog'); const box = await dialog.boundingBox(); // Should not exceed viewport (414px) expect(box?.width).toBeLessThanOrEqual(414); }); test('no horizontal overflow', async ({ page }) => { await page.goto('/settings'); const scrollWidth = await page.evaluate(() => document.documentElement.scrollWidth); const clientWidth = await page.evaluate(() => document.documentElement.clientWidth);
